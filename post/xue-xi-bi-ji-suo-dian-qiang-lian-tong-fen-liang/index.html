这是针对有向图的一个概念。*强连通分量（scc）*：一个强连通分量之间的点可以相互到达。

我们可以用 tarjan 算法计算一个有向图的强连通分量。

tarjan 引入了两个数组，巧妙地解决了这个问题。

第一个数组 dfnxdfn_xdfnx​ 表示时间戳，意思就是 xxx 是 dfs 序中第几个被访问的节点。

第二个数组 lowxlow_xlowx​ 表示不经过父亲节点可以访问到的 dfndfndfn 的最小值。

那么怎么去求强连通分量呢？

我们考虑，对于一个 scc 中，我们肯定只有一个 dfnx=lowxdfn_x = low_xdfnx​=lowx​ 的点。而且 scc 中所有的点的 dfndfndfn 肯定都大于这个点，那么我们可以设计一个栈，每次都把栈弹到 dfnx=lowxdfn_x=low_xdfnx​=lowx​ 的地方，这样的话其中的所有点肯定是属于这个scc 内部的。

void tarjan(int u) {
    dfn[u] = low[u] = ++cnt;
    sta[++top] = u; vis[u] = 1;
    instack[u] = 1;
    for(int v : G[u]) {
        if(!vis[v]) {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        } else if(instack[v]) low[u] = min(low[u], dfn[v]);
    }
    if(low[u] == dfn[u]) {
        ++ss;
        do {
            instack[sta[top]] = 0;
            b[ss].push_back(sta[top]);
            own[sta[top]] = ss;
        }while(sta[top--] != u);
    }
}

tarjan 算法，还可以用来求割点。

很直观的，我们能想到，如果对于一个 dfs 树，某一个节点的子节点中有一个满足“不经过父亲能到达的节点都在父节点为根节点的子树里”的节点，那么父节点就是一个割点。

那么类似于上面的 tarjan 算法，如果一个不是根节点的点（因为根节点没有父亲，判断方式不一样），他的儿子中存在一个 low 值比我 dfn 低的，那么这个点就是割点。

那么对于根节点，如果我有两个及以上的儿子，那么这个点肯定是根节点，这是很显然的。

void tarjan(int u, bool root) {
    dfn[u] = low[u] = ++cnt;
    vis[u] = 1;
    int ch = 0;
    for(int v : G[u]) {
        if(!vis[v]) {
            tarjan(v, 0);
            if(low[v] >= dfn[u] && !root) isg[u] = 1;
            if(root) ++ch;
            low[u] = min(low[u], low[v]);
        } else low[u] = min(low[u], dfn[v]);
    }
    if(root && ch >= 2) isg[u] = 1;
}

点双连通分量
什么叫点双连通分量呢，就是不存在割点的图。很直观的，我们只需要将割点分成图的多个部分提取就行了。

void tarjan(int u, bool root) {
    sta[++top] = u;
    dfn[u] = low[u] = ++cnt;
    vis[u] = 1;
    int ch = 0;
    for(int v : G[u]) {
        if(!vis[v]) {
            ++ch;
            tarjan(v, 0);
            low[u] = min(low[u], low[v]);
            if(low[v] >= dfn[u]) {
                ++s;
                while(sta[top + 1] != v) ds[s].push_back(sta[top--]);
                ds[s].push_back(u);
            }
        } else low[u] = min(dfn[v], low[u]);
    }
    if(root && ch == 0) ds[++s].push_back(u);
}

边双连通分量
类似于点双连通分量，我们如果把割边舍弃，剩下的部分就是边双连通分量啦。

#include<bits/stdc++.h>
using namespace std;
int n, m;
const int N = 5e5 + 10, M = 4e6 + 10;
int head[N];
struct Edge {
    int v, nxt;
}e[M];
int cnt = 1, dfn[N], low[N], s;
bool vis[N], g[M];
void add(int x, int y) {
    e[++cnt] = {y, head[x]};
    head[x] = cnt;
}
void tarjan(int u, int fr) {
    dfn[u] = low[u] = ++s; vis[u] = 1;
    for(int i = head[u]; i; i = e[i].nxt) {
        int v = e[i].v;
        if(!vis[v]) {
            tarjan(v, i);
            if(dfn[u] < low[v]) g[i] = g[i ^ 1] = 1;
            low[u] = min(low[u], low[v]);
        } else if(i != (fr ^ 1)) low[u] = min(low[u], dfn[v]);
    }
}
vector<int>b[N];
int ans;
void dfs(int u) {
    vis[u] = 1; 
    b[ans].push_back(u);
    for(int i = head[u]; i; i = e[i].nxt) {
        if(g[i]) continue;
        int v = e[i].v;
        if(!vis[v]) dfs(v);
    }
}
int main() {
    cin >> n >> m;
    for(int i = 1; i <= m; ++i) {
        int x, y;
        cin >> x >> y;
        add(x, y);
        add(y, x);
    }
    for(int i = 1; i <= n; ++i) if(!vis[i]) tarjan(i, 0);
    memset(vis, 0, sizeof(vis));
    for(int i = 1; i <= n; ++i) {
        if(!vis[i]) {
            ++ans;
            dfs(i);
        }
    }
    cout << ans << endl;
    for(int i = 1; i <= ans; ++i) {
        cout << b[i].size() << ' ';
        for(int u : b[i]) cout << u << ' ';
        cout << endl;
    }
    return 0;
}
