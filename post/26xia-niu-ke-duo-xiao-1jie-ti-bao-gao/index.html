C Fish Eating

为了这道题专门去学了一下 Kruskal 重构树。
这个题整个就很像一个 Kruskal 重构树。从小到大的加入鱼，我们去用并查集的话还可以满足这是一个大根堆。（并查集本身是一棵树嘛）然后我们维护如果要网上走要达到的值，这个值应该是 vafa−szu+1va_{fa}-sz_{u}+1vafa​−szu​+1 。这很显然嘛。因为在整个过程中我一共可以吃 szu−1sz_u-1szu​−1 条鱼（不包括自己）。然后还要达到 vafava_{fa}vafa​ 的大小，那么就是这两个相减喽。
好的，然后我们考虑操作 222 的话，就是要求想求点到根的要达到的值的最大值。我们发现这个值是可以并查集维护的（需要思考）然后那么就可以做啦~

#include<bits/stdc++.h>
using namespace std;
int n, m, q, l;
const int dx[] = {1, -1, 0, 0},
          dy[] = {0, 0, 1, -1};
const int N = 1e6 + 10;
int fa[N], sz[N], mx[N];
int findf(int x) {
    if(fa[x] == x) return x;
    int y = findf(fa[x]);
    mx[x] = max(mx[x], mx[fa[x]]);
    return fa[x] = y;
}
int a[N];

int main() {
    cin >> n >> m >> q;
    for(int i = 1; i <= n * m; ++i) fa[i] = i, sz[i] = 1, mx[i] = 0;
    while(q--) {
        int opt, x, y, v;
        cin >> opt;
        if(opt == 1) {
            cin >> x >> y >> v;
            x ^= l; y ^= l;
            a[(x - 1) * m + y] = v;
            for(int d = 0; d < 4; ++d) {
                int nx = x + dx[d], ny = y + dy[d];
                if(nx >= 1 && ny >= 1 && nx <= n && ny <= m) {
                    int id = (nx - 1) * m + ny;
                    if(a[id]) {
                        int fu = findf((x - 1) * m + y), fv = findf(id);
                        if(fu == fv) continue;
                        sz[fu] += sz[fv];
                        mx[fv] = max(a[fu] - sz[fv] + 1, mx[fv]);
                        // cout << fu << ' ' << fv << ' ' << mx[fv] << ' ' << a[fv] << ' ' << a[fu] << ' ' << sz[fv] << endl;
                        fa[fv] = fu;
                    }
                }
            }
            cout << (l = sz[findf((x - 1) * m + y)] - 1) << endl;
        } else {
            int x, y;
            cin >> x >> y;
            x ^= l; y ^= l;
            // cout << x << ' ' << y << endl;
            findf((x - 1) * m + y);
            cout << (l = (max(0, mx[(x - 1) * m + y] - a[(x - 1) * m + y]))) << endl;
        }
    }
    return 0;
}

H Rock-Paper-Scissors Master

马尔科夫链的题，重点是想到马尔科夫链的一个性质，后面不同状态的概率趋于稳定。这题里就是一样的，于是后面的期望相邻之差趋于一个等差数列，我们发现 K=104K = 10^4K=104 的时候已经非常收敛，可以通过本题。
包括这种题目是怎么推的。

#include<bits/stdc++.h>
using namespace std;
typedef long double db;
int T;
struct Cards {
    int c[4];
};
const int N = 1e4 + 10;
Cards rev[15];
int idx, id_map[5][5][5], nxt[105][4][4][4][4];
db dp[N][11][11];
db eps = 1e-12;
int score(int x, int y) {
    if((x == 1 && y == 2) || (x == 2 && y == 3) || (x == 3 && y == 1)) return 3;
    if(x == y) return 1;
    return 0;
}
db avr = 0;
bool vis[N][11][11];
db dfs(int dep, int stat1, int stat2) {
    if(dep == 0) return 0;
    if(vis[dep][stat1][stat2]) return dp[dep][stat1][stat2];
    vis[dep][stat1][stat2] = 1;
    db maxx = 0;
    for(int k = 1; k <= 3; ++k) {
        if(rev[stat1].c[k] == 0) continue;
        db minn = 1e18;
        for(int l = 1; l <= 3; ++l) {
            if(rev[stat2].c[l] == 0) continue;
            db summ = 0;
            for(int i = 1; i <= 3; ++i) {
                for(int j = 1; j <= 3; ++j) {
                    Cards new1 = rev[stat1], new2 = rev[stat2];
                    new1.c[i] ++, new1.c[k] --;
                    new2.c[j] ++, new2.c[l] --;
                    summ += dfs(dep - 1, id_map[new1.c[1]][new1.c[2]][new1.c[3]], id_map[new2.c[1]][new2.c[2]][new2.c[3]]);
                }
            }
            summ /= 9;
            summ += score(k, l);
            minn = min(minn, summ);
        }
        maxx = max(maxx, minn);
    }
    return dp[dep][stat1][stat2] = maxx;
}
void intt() {
    for(int i = 0; i <= 3; ++i) {
        for(int j = 0; j <= 3 - i; ++j) {
            int k = 3 - i - j;
            id_map[i][j][k] = ++idx;
            rev[idx].c[1] = i, rev[idx].c[2] = j, rev[idx].c[3] = k;
        }
    }
    for(int stat1 = 1; stat1 <= idx; ++stat1)
        for(int stat2 = 1; stat2 <= idx; ++stat2)
            dfs(N - 1, stat1, stat2);
    for(int stat1 = 1; stat1 <= idx; ++stat1)
        for(int stat2 = 1; stat2 <= idx; ++stat2)
            avr += dp[N - 1][stat1][stat2] - dp[N - 2][stat1][stat2];
    // cout << avr << endl;
    avr /= 100;
}
void work() {
    int k;
    cin >> k;
    string st1, st2;
    cin >> st1 >> st2;
    Cards A, B;
    memset(A.c, 0, sizeof A.c); memset(B.c, 0, sizeof B.c);
    for(int i = 0; i < st1.size(); ++i) {
        if(st1[i] == 'R') A.c[1] ++;
        if(st1[i] == 'S') A.c[2] ++;
        if(st1[i] == 'P') A.c[3] ++;
    }
    for(int i = 0; i < st2.size(); ++i) {
        if(st2[i] == 'R') B.c[1] ++;
        if(st2[i] == 'S') B.c[2] ++;
        if(st2[i] == 'P') B.c[3] ++;
    }
    db ans = dp[min(N - 1, k)][id_map[A.c[1]][A.c[2]][A.c[3]]][id_map[B.c[1]][B.c[2]][B.c[3]]];
    if(k >= N) {
        ans += avr * (k - (N - 1));
    }
    cout << fixed << setprecision(9) << ans << endl;
}
int main() {
    ios::sync_with_stdio(), cin.tie();
    cin >> T;
    intt();
    while(T--) work();
    return 0;
}
