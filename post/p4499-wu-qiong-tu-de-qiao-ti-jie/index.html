P4499 无穷图的桥 题解

2026-07-13

前置知识：缩点 / 边双连通分量
边双在这题里十分重要是很显然的一点，我们考虑每一层的边双连通分量和连通分量。

我们定义，同一层中的边叫做横边，其他边叫竖边。

我们考虑一下如果一条边，他要么一直是桥，要么在前 iii 层是桥，之后就不是了。

于是我们只需要考虑每条边什么时候开始不是桥。

我们再说一下如何去看这张图，我们可以把最小的“像素点”（最小节点）看成边双，因为边双内部的结构不再重要，其中的边不可能再次成为桥。然后这些边双构成森林，一棵树相当于一个较大的连通分量。

我们证明一个引理：如果这一层的一个连通分量含有两条竖边，那么在下一层中，连出去的这些竖边将把连出去的点合成为一个边双。

我们考虑这样一个情况。记第 iii 层的点集为 TiT_iTi​，总边集为 EEE。

若 ui,vi,xi,yi∈Tiu_i,v_i,x_i,y_i\in T_iui​,vi​,xi​,yi​∈Ti​ 如果 uuu 和 vvv 属于同一个连通分量。 (ui,xi+1),(vi,yi+1)∈E(u_i,x_{i+1}), (v_i,y_{i+1})\in E(ui​,xi+1​),(vi​,yi+1​)∈E 那么 uiu_iui​ 到 viv_ivi​ 这个边双树上路径缩为一个边双，并且下一层中，将 xi+1x_{i+1}xi+1​ 和 yi+1y_{i+1}yi+1​ 连边。

为什么呢？

[注意下图中一个点表示一个连通分量，这一点有别于上面我们看这个图的方式。]

由于这个图是同构的，也就是如果我们前 iii 层遮住，那么后面的图仍然与原图同构（不考虑边权）。那么于是上图中，如果我们把第一行遮住，由于该图与原图同构，并且原图连通，那么 xi+1,yi+1x_{i+1}, y_{i+1}xi+1​,yi+1​ 必然是和 ui+1u_{i+1}ui+1​ 和 vi+1v_{i+1}vi+1​ 的这个连通分量是通过后面层的某种方式相连接的。那么我们可以合并 uiu_iui​ 到 viv_ivi​ 在边双树上的路径经过的边双。

之所以给 xi+1x_{i+1}xi+1​ 和 yi+1y_{i+1}yi+1​ 连边是因为我们可以想到 uiu_iui​ 到 viv_ivi​ 的这个边双
（已经并成一个边双了嘛）给 xi+1x_{i+1}xi+1​ 和 yi+1y_{i+1}yi+1​ 提供了一条相互到达的路，那么可以等效于给 xi+1x_{i+1}xi+1​ 和 yi+1y_{i+1}yi+1​ 连了一条边。

那么可以想到合并的次数不会太多（O(n)\mathcal{O}(n)O(n)）的。那么我们已经大概知道了我们需要做的过程。

后面需要补充一点细节。

考虑一下连边过程，如果 xi+1x_{i+1}xi+1​ 和 yi+1y_{i+1}yi+1​ 属于同一个连通分量怎么办？那么我们可以想到，这构成了一个环，我们需要把他们继续合并起来。
那么如果他们不属于同一个连通分量呢？那么其实就是把两个连通块合并了起来，而由于每一个连通块至少有一个竖边（不然的话这个连通块孤立了就不可能整张图连通了呀）。那么这个新合并成的连通块有了两个竖边，你会发现他还能继续合并。

好了，这就是这题的全部做法。本题的实现难度也很高。

#include<bits/stdc++.h>
using namespace std;
typedef long double ldb;
const int N = 1e6 + 10, M = 4e6 + 10;
int n, m1, m2;
ldb p[N];
struct Edge {
    int v, val, nxt;
}e[M];
int Ecnt = 1, head[N];
void add(int x, int y, int val) {
    e[++Ecnt] = {y, val, head[x]};
    head[x] = Ecnt;
}
vector<pair<int, int> >G[N];
int dfn[N], low[N], Tcnt, U[N], V[N], VAL[N];
bool vis[N], b[M];
void tarjan(int u, int fr) {
    dfn[u] = low[u] = ++Tcnt;
    vis[u] = 1;
    for(int i = head[u]; i; i = e[i].nxt) {
        int v = e[i].v;
        if(!vis[v]) {
            tarjan(v, i);
            if(low[v] > dfn[u]) b[i] = b[i ^ 1] = 1;
            low[u] = min(low[u], low[v]);
        } else if(i != (fr ^ 1)) low[u] = min(dfn[v], low[u]);
    }
}
int Bcnt, id[N];
int w[N];
void dfs1(int u) {
    vis[u] = 1; id[u] = Bcnt;
    for(int i = head[u]; i; i = e[i].nxt) {
        if(b[i]) continue;
        int v = e[i].v;
        if(!vis[v]) {
            dfs1(v);
        }
    }
}
struct Dsu {
    int fa[N];
    void intt(int X) {
        for(int i = 1; i <= X; ++i) fa[i] = i;
    }
    int findf(int x) {
        return (fa[x] == x) ? x : (fa[x] = findf(fa[x]));
    }
    void merge(int x, int y) {
        int X = findf(x), Y = findf(y);
        if(X ^ Y) fa[X] = Y;
    }
}dsu1, dsu2;
int demo[N], tim[N], fa[N], dep[N], xh[N], rt[N];
void dfs(int u, int Fa) {
    dep[u] = dep[Fa] + 1; fa[u] = Fa;
    for(auto [v, uid] : G[u]) {
        if(v == Fa) continue;
        xh[v] = uid;
        dfs(v, u);
    }
}
void chkmint(int x, int y) {
    tim[x] = min(tim[x], y);
}
void access(int T, int y) {
    while(y) {
        y = dsu2.findf(y);
        if(fa[y] == 0) return;
        chkmint(xh[y], T);
        dsu2.merge(y, fa[y]);
    }
}
bool used[M];
int main() {
    ios::sync_with_stdio(), cin.tie();
    p[0] = 1; dep[0] = 0;
    for(int i = 1; i < N; ++i) p[i] = p[i - 1] * 0.9;
    for(int i = 1; i < N; ++i) p[i] += p[i - 1];
    cin >> n >> m1 >> m2;
    for(int i = 1; i <= m1; ++i) {
        int x, y, val;
        cin >> x >> y >> val;
        add(x, y, val);
        add(y, x, val);
        w[i] = val;
    }
    for(int i = 1; i <= m2; ++i) {
        cin >> U[i] >> V[i] >> VAL[i];
        w[i + m1] = VAL[i];
    }
    for(int i = 1; i <= n; ++i) if(!vis[i]) tarjan(i, 0);
    memset(vis, 0, sizeof(vis));
    memset(tim, 0x3f, sizeof(tim));
    for(int i = 1; i <= n; ++i) if(!vis[i]) ++Bcnt, dfs1(i);
    dsu1.intt(Bcnt);
    for(int i = 1; i <= n; ++i) {
        for(int j = head[i]; j; j = e[j].nxt) {
            int v = e[j].v;
            int eid = (min(j, j ^ 1) + 1) / 2;
            if(used[eid]) continue;
            used[eid] = 1;
            if(dsu1.findf(id[i]) != dsu1.findf(id[v])) {
                G[id[i]].push_back({id[v], eid});
                G[id[v]].push_back({id[i], eid});
                dsu1.merge(id[i], id[v]);
            } else chkmint(eid, -1);
        }
    }
    queue<array<int, 3> >Q;
    for(int i = 1; i <= m2; ++i) {
        int t = dsu1.findf(id[U[i]]);
        if(!demo[t]) demo[t] = i, rt[t] = id[U[i]];
        else {
            Q.push({id[V[demo[t]]], id[V[i]], 0});
        }
    }
    for(int i = 1; i <= Bcnt; ++i) 
        if(dsu1.fa[i] == i) dfs(rt[i], 0);
    dsu2.intt(Bcnt);
    for(int i = 1; i <= m2; ++i) {
        int t = dsu1.findf(id[U[i]]);
        if(demo[t] != i) access(-1, id[U[i]]), chkmint(i + m1, -1), chkmint(demo[t] + m1, -1);
    }
    while(!Q.empty()) { 
        auto it = Q.front(); Q.pop();
        int u = it[0], v = it[1], val = it[2];
        int uu = dsu1.findf(u), vv = dsu1.findf(v);
        if(uu == vv) {
            uu = dsu2.findf(u), vv = dsu2.findf(v);
            while(uu ^ vv) {
                if(dep[uu] < dep[vv]) swap(uu, vv);
                chkmint(xh[uu], val);
                dsu2.merge(uu, fa[uu]);
                uu = dsu2.findf(uu);
            }
        } else {
            uu = dsu2.findf(u), vv = dsu2.findf(v);
            access(val, uu);
            access(val, vv);
            uu = dsu1.findf(uu), vv = dsu1.findf(vv);
            dsu1.merge(uu, vv);
            dsu2.fa[rt[uu]] = rt[vv]; 
            chkmint(demo[uu] + m1, val);
            chkmint(demo[vv] + m1, val);
            Q.push({id[V[demo[uu]]], id[V[demo[vv]]], val + 1});
        }
    }
    long double ans = 0;
    for(int i = 1; i <= m1 + m2; ++i) {
        if(tim[i] == -1) continue;
        if(tim[i] < 1e9) ans += p[tim[i]] * w[i];
        else ans += 10 * w[i];
    }
    cout << fixed << setprecision(2) << ans << endl;
    return 0;
}
